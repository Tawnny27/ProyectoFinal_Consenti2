import React, { useState } from 'react';
import axios from 'axios';

const RegistrarUsuarioInterno = () => {
  const [usuario, setUsuario] = useState({
    idUsuario: 0,
    rolId: 0,
    nombreUsuario: '',
    apellidosUsuario: '',
    cedulaUsuario: '',
    telefonoUsuario: '',
    correoUsuario: '',
    rol: {
      idRol: 0,
      nombreRol: '',
      usuariosInternos: [''],
      padres: [
        {
          idPadre: 0,
          rolId: 0,
          nombrePadre: '',
          apellidosPadre: '',
          cedulaPadre: '',
          telefonoPadre: '',
          correoPadre: '',
          rol: '',
          hijos: [
            {
              idAlumno: 0,
              padreId: 0,
              nombreAlumno: '',
              apellidosAlumno: '',
              fechaNacimiento: '',
              cedulaAlumno: '',
              generoAlumno: '',
              direccionAlumno: '',
              informacionAdicional: '',
              fotoAlumno: '',
              nombreCompAutorizado: '',
              cedulaAutorizado: '',
              telefonoAutorizado: '',
              relacionAutorizado: '',
              nombreCompContacto: '',
              cedulaContacto: '',
              telefonoContacto: '',
              relacionContacto: '',
            },
          ],
        },
      ],
    },
  });

  const handleChange = (e) => {
    const { name, value } = e.target;
    setUsuario({ ...usuario, [name]: value });
  };

  const handleRolChange = (e) => {
    const { name, value } = e.target;
    setUsuario({
      ...usuario,
      rol: { ...usuario.rol, [name]: value },
    });
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    axios
      .post('URL_DEL_BACKEND', usuario)
      .then((response) => {
        console.log('Usuario registrado:', response.data);
      })
      .catch((error) => {
        console.error('Error registrando usuario:', error);
      });
  };

  return (
    <form onSubmit={handleSubmit}>
      <h2>Registrar Usuario Interno</h2>
      <input
        type="text"
        name="nombreUsuario"
        value={usuario.nombreUsuario}
        onChange={handleChange}
        placeholder="Nombre del usuario"
        required
      />
      <input
        type="text"
        name="apellidosUsuario"
        value={usuario.apellidosUsuario}
        onChange={handleChange}
        placeholder="Apellidos del usuario"
        required
      />
      <input
        type="text"
        name="cedulaUsuario"
        value={usuario.cedulaUsuario}
        onChange={handleChange}
        placeholder="Cédula del usuario"
        required
      />
      <input
        type="tel"
        name="telefonoUsuario"
        value={usuario.telefonoUsuario}
        onChange={handleChange}
        placeholder="Teléfono del usuario"
        required
      />
      <input
        type="email"
        name="correoUsuario"
        value={usuario.correoUsuario}
        onChange={handleChange}
        placeholder="Correo electrónico del usuario"
        required
      />

      <h3>Información del Rol</h3>
      <input
        type="text"
        name="nombreRol"
        value={usuario.rol.nombreRol}
        onChange={handleRolChange}
        placeholder="Nombre del rol"
        required
      />
      {/* Puedes agregar más campos para capturar detalles del rol según sea necesario */}

      <button type="submit">Registrar</button>
    </form>
  );
};

export default RegistrarUsuarioInterno;
