import React, { useState } from "react";

const FormularioUsuarioInterno = () => {
  const [usuarioInterno, setUsuarioInterno] = useState({
    idUsuario: 0,
    rolId: 0,
    nombreUsuario: "",
    apellidosUsuario: "",
    cedulaUsuario: "",
    telefonoUsuario: 0,
    correoUsuario: "",
  });

  const handleChange = (e) => {
    const { name, value } = e.target;
    setUsuarioInterno({ ...usuarioInterno, [name]: value });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      const response = await fetch("http://tu-api-url/CrearUsuarioInterno", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(usuarioInterno),
      });
      if (response.ok) {
        alert("Usuario creado exitosamente");
        // Puedes limpiar el formulario o redirigir a otro lado si es necesario
      } else {
        alert("Error al crear el usuario");
      }
    } catch (error) {
      console.error("Error:", error);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="nombreUsuario">Nombre:</label>
        <input
          type="text"
          id="nombreUsuario"
          name="nombreUsuario"
          value={usuarioInterno.nombreUsuario}
          onChange={handleChange}
          required
        />
      </div>
      <div>
        <label htmlFor="apellidosUsuario">Apellidos:</label>
        <input
          type="text"
          id="apellidosUsuario"
          name="apellidosUsuario"
          value={usuarioInterno.apellidosUsuario}
          onChange={handleChange}
          required
        />
      </div>
      <div>
        <label htmlFor="cedulaUsuario">Cédula:</label>
        <input
          type="text"
          id="cedulaUsuario"
          name="cedulaUsuario"
          value={usuarioInterno.cedulaUsuario}
          onChange={handleChange}
          required
        />
      </div>
      <div>
        <label htmlFor="telefonoUsuario">Teléfono:</label>
        <input
          type="number"
          id="telefonoUsuario"
          name="telefonoUsuario"
          value={usuarioInterno.telefonoUsuario}
          onChange={handleChange}
          required
        />
      </div>
      <div>
        <label htmlFor="correoUsuario">Correo:</label>
        <input
          type="email"
          id="correoUsuario"
          name="correoUsuario"
          value={usuarioInterno.correoUsuario}
          onChange={handleChange}
          required
        />
      </div>
      <div>
        <label htmlFor="rolId">Rol:</label>
        <input
          type="number"
          id="rolId"
          name="rolId"
          value={usuarioInterno.rolId}
          onChange={handleChange}
          required
        />
      </div>
      <button type="submit">Crear Usuario</button>
    </form>
  );
};

export default FormularioUsuarioInterno;
